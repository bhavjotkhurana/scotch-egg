\section*{Practice Set}
\begin{enumerate}
  \item Solve for $x$: $2x-7=11$.
  \item If a line has slope $-\tfrac{3}{2}$ and passes through $(4,5)$, write its equation in slope-intercept form.
  \item A system has equations $y=2x+3$ and $y=-x+9$. Find the solution.
\end{enumerate}

\section*{Answer Key (Preview)}
\begin{enumerate}
  \item $x=9$
  \item $y=-\tfrac{3}{2}x+11$
  \item Intersection: $x=2$, $y=7$.
\end{enumerate}
